        <input type="text" id="gpas" placeholder="e.g., 3.2, 3.4, 3.5">

        

        <button onclick="calculateGPA()">Calculate Minimum GPA</button>

        

        <div class="output" id="output"></div>

    </div>


    <script>

        function calculateGPA() {

            // Constants

            const totalSemesters = 8;

            const unitsPerSemester = 18;

            const requiredGPA = 3.6;

            const totalUnits = totalSemesters * unitsPerSemester;

            const totalGradePointsRequired = requiredGPA * totalUnits;


            // Get user input

            const gpaInput = document.getElementById('gpas').value;

            const semesterGPAs = gpaInput.split(',').map(gpa => parseFloat(gpa.trim()));


            // Validate input

            if (semesterGPAs.some(isNaN)) {

                document.getElementById('output').innerHTML = "<p style='color: red;'>Please enter valid GPA values separated by commas.</p>";

                return;

            }


            // Calculate grade points earned

            const unitsCompleted = semesterGPAs.length * unitsPerSemester;

            const gradePointsEarned = semesterGPAs.reduce((sum, gpa) => sum + gpa * unitsPerSemester, 0);


            // Calculate remaining grade points and units

            const remainingGradePoints = totalGradePointsRequired - gradePointsEarned;

            const remainingUnits = totalUnits - unitsCompleted;


            // Calculate minimum GPA needed for remaining semesters

            let result;

            if (remainingUnits > 0) {

                const gpaNeededPerSemester = remainingGradePoints / remainingUnits;

                result = `

                    <p><strong>Grade Points Earned So Far:</strong> ${gradePointsEarned.toFixed(2)}</p>

                    <p><strong>Remaining Grade Points Needed:</strong> ${remainingGradePoints.toFixed(2)}</p>

                    <p><strong>Remaining Units:</strong> ${remainingUnits}</p>

                    <p><strong>Minimum GPA Needed for Remaining Semesters:</strong> ${gpaNeededPerSemester.toFixed(2)}</p>

                `;

            } else {

                result = "<p>All semesters completed. No GPA calculation needed.</p>";

            }


            // Display result

            document.getElementById('output').innerHTML = result;

        }

    </script>
